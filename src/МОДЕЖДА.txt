if $ARGS[0] = '1': 
cla
$metka = $ARGS[0]
$loc = $CURLOC
'<center><img src = "pic\manOd\kostum.jpg" ></center>'
'Классический мужской костюм.'
act 'Убрать вещь.': GOTO $locP, $metkaP
     if kostum = 0 and mag = 1: 
     'Цена <<payclo>>'
         if money >= payclo: 
			 act 'Купить': 
			 cla
			  money = money - payclo
			  kostum = 1
			 ! kostumB = bedra
			  kostumE = 5000
			 'Вы заплатили на кассе <<payclo>> руб и забрали покупку'
			 act 'Уйти.': GOTO $locP, $metkaP
			 end
		 end
	 elseif mag = 0: 
		 if kostumE > 0: 
			 act 'Надеть': 
			 cla
			 gs 'clonil'
			  slot = 1
			  kostum = 2
			 GOTO $locP, $metkaP
			 end
		 end
		 
		 if kostumE <= 0: 
		  kostum = 0
		 XGOTO $locP, $metkaP
		 'Костюм слишком изношен, поэтому пришлось его выбросить.'
		 end
	 
     end
end
