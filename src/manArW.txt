if $ARGS[0]='1':
cla
$metka = $ARGS[0]
$loc = $CURLOC
if tits<=0:'<center><img src="pic/EvedonOd/armor.jpg" ></center>'
if tits>0:'<center><img src="pic/EvedonOd/armorf.jpg" ></center>'
'Легкий доспех.'
'Защита 50'
act 'Убрать вещь.':GOTO $locP, $metkaP
     if armor = 0 and mag = 1:
     'Цена <<payclo>>'
         if gem >= payclo:
			 act 'Купить':
			 cla
			 set gem = gem - payclo
			 set armor = 1
			 !set kostumB = bedra
			 'Вы заплатили <<payclo>> гемов и забрали покупку'
			 act 'Уйти.':GOTO $locP, $metkaP
			 end
		 end
	 elseif mag = 0:
			 act 'Надеть':
			 cla
			 gs'clonil'
			 set slot = 100
			 set armor = 2
			 armorDef=50
			 GOTO $locP, $metkaP
			 end
     end
end
