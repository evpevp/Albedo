if $ARGS[0]='win':
     if FightTip = 0:
     set WIN = WIN + 1
     end
     if FightTip = 1:
         if amature = 0:set BOXWIN = BOXWIN + 1
         if amature = 1:set BOXWINp = BOXWINp + 1
     end
     if FightTip = 2:
         if amature = 0:set KIKWIN = KIKWIN + 1
         if amature = 1:set KIKWINp = KIKWINp + 1
     end
     if FightTip = 3:
     set WRESLWIN = WRESLWIN + 1
     end
     if FightTip = 4:
     set MMAWIN = MMAWIN + 1
     end
gs'FightNill'
act 'Уйти.':GOTO $locFight, $metkaFight
end
