koefspeed = speedv*20/100
koefagil = agil*20/100
pop = RAND(speedv-koefspeed,speedv+koefspeed)
uvo = RAND((agil-Kar)-koefagil,(agil-Kar)+koefagil)

if pop>= uvo:

end
if pop<uvo: 
'<B><font color = green>ВЫ УВЕРНУЛИСЬ ОТ ПОДСЕЧКИ</font></B>'
end
