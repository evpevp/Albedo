cla

$loc = $CURLOC
if tits<=0:'<center><img src="pic/EvedonOd/roba.jpg" ></center>'

'Рубище.'
act 'Убрать вещь.':GOTO $locP, $metkaP
     if roba = 0 and mag = 1:
     'Цена <<payclo>>'
         if gem >= payclo:
			 act 'Купить':
			 cla
			 set gem = gem - payclo
			 set roba = 1
			 !set kostumB = bedra
			 'Вы заплатили <<payclo>> руб и забрали покупку'
			 act 'Уйти.':GOTO $locP, $metkaP
			 end
		 end
	 elseif mag = 0:
			 act 'Надеть':
			 cla
			 gs'clonil'
			 set slot = 11
			 set roba = 2
			 GOTO $locP, $metkaP
			 end
     end
end
